import os
import csv

def extract_image_filenames(folder_path):
    image_files = [f for f in os.listdir(folder_path) if os.path.isfile(os.path.join(folder_path, f)) and f.lower().endswith(('.png', '.jpg', '.jpeg', '.gif', '.bmp', '.tiff'))]
    return image_files

def save_to_csv(image_files, csv_filename):
    with open(csv_filename, 'w', newline='') as csvfile:
        csv_writer = csv.writer(csvfile)
        csv_writer.writerow(['Image Filename'])
        for filename in image_files:
            csv_writer.writerow([filename])

if __name__ == "__main__":
    folder_path = input("Enter the folder path: ")
    csv_filename = input("Enter the CSV filename to save: ")

    image_files = extract_image_filenames(folder_path)
    save_to_csv(image_files, csv_filename)
    print("Image filenames saved to", csv_filename)
