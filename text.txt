    <div class="container mt-5">
        <h2>JSON Table Editor</h2>
        <table class="table table-bordered mt-3" id="jsonTable">
            <thead>
                <tr>
                    <th>ID</th>
                    <th>Name</th>
                    <th>Age</th>
                </tr>
            </thead>
            <tbody>
                {% for entry in data %}
                    <tr>
                        <td>{{ entry.id }}</td>
                        <td>{{ entry.name }}</td>
                        <td>{{ entry.age }}</td>
                    </tr>
                {% endfor %}
            </tbody>
        </table>
        <button class="btn btn-primary mt-3" onclick="editTable()">Edit Table</button>
        <button class="btn btn-success mt-3" onclick="saveChanges()">Save Changes</button>
    </div>

    <script src="https://cdn.jsdelivr.net/npm/bootstrap@5.3.0/dist/js/bootstrap.bundle.min.js"></script>
    <script>
        function editTable() {
            // Enable editing of the table cells
            document.querySelectorAll('#jsonTable td').forEach(cell => {
                cell.contentEditable = true;
                cell.style.border = '1px solid #ccc';
            });
        }

        function saveChanges() {
            // Collect the updated data from the table
            var updatedData = [];
            document.querySelectorAll('#jsonTable tbody tr').forEach(row => {
                var entry = {
                    id: row.cells[0].innerText,
                    name: row.cells[1].innerText,
                    age: row.cells[2].innerText
                };
                updatedData.push(entry);
            });

            // Send the updated data to the server
            fetch('/update_table', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                },
                body: JSON.stringify(updatedData),
            })
            .then(response => response.json())
            .then(data => {
                if (data.status === 'success') {
                    alert('Changes saved successfully!');
                } else {
                    alert('Error saving changes.');
                }
            });
        }
    </script>
</body>
</html>
