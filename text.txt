import os
import json

def modify_json_data(input_folder, output_folder):
    # Create the output folder if it doesn't exist
    if not os.path.exists(output_folder):
        os.makedirs(output_folder)

    # Iterate over all files in the input folder
    for filename in os.listdir(input_folder):
        if filename.endswith(".json"):
            # Read the JSON file
            with open(os.path.join(input_folder, filename), 'r') as file:
                json_data = json.load(file)

            # Modify the JSON data (Replace this with your own logic)
            modified_json_data = {key: value.upper() if isinstance(value, str) else value for key, value in json_data.items()}

            # Write the modified JSON data to a new file in the output folder
            output_filename = os.path.join(output_folder, filename)
            with open(output_filename, 'w') as file:
                json.dump(modified_json_data, file, indent=4)

            print(f"Modified JSON saved to: {output_filename}")

# Example usage:
input_folder = "/path/to/input/folder"
output_folder = "/path/to/output/folder"
modify_json_data(input_folder, output_folder)
