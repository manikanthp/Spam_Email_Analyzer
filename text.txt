import cv2
import numpy as np

def classify_intensity_variation(image_path):
    # Load the image
    img = cv2.imread(image_path, cv2.IMREAD_GRAYSCALE)

    # Get the image dimensions
    height, width = img.shape

    # Split the image into top and bottom halves
    top_half = img[:height//2, :]
    bottom_half = img[height//2:, :]

    # Calculate the mean intensity for each half
    mean_intensity_top = np.mean(top_half)
    mean_intensity_bottom = np.mean(bottom_half)

    # Set a threshold for intensity difference
    intensity_threshold = 20

    # Classify based on intensity difference
    if abs(mean_intensity_top - mean_intensity_bottom) > intensity_threshold:
        return "Top part has different intensity"
    else:
        return "Bottom part has different intensity"

# Example usage
image_path = "path/to/your/image.jpg"
result = classify_intensity_variation(image_path)
print(result)
